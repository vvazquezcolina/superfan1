export const CitySectionSchema = z.object({
  title: LocalizedText,
  content: LocalizedText, // HTML-free markdown-style paragraphs, separated by \n\n
})

export const CityFAQSchema = z.object({
  question: LocalizedText,
  answer: LocalizedText,
})

export const CitySourceSchema = z.object({
  name: z.string().min(1),
  url: z.string().url(),
})

export const CityContentSchema = z.object({
  overview: LocalizedText,           // 2-3 paragraphs, the "hook" — first 200 words answer "why visit this city for the World Cup?"
  gettingThere: CitySectionSchema,   // Flights, land transport from other host cities
  gettingAround: CitySectionSchema,  // Metro, buses, ride-share, taxis, walkability
  neighborhoods: CitySectionSchema,  // Where to stay, 3-4 neighborhood recs with character description
  foodAndDrink: CitySectionSchema,   // Local cuisine, must-try dishes, budget-friendly + splurge options
  safety: CitySectionSchema,         // Honest practical safety tips, areas to avoid, emergency numbers
  weather: CitySectionSchema,        // June-July weather specifically, what to pack
  culturalContext: CitySectionSchema, // Per D-02/CITY-02: Spanish-speaking services, Latino community, consulates, gathering spots

})
