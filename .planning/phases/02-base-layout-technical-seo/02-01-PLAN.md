```typescript
export default async function LangLayout({ children, params }) {
  const { lang } = await params
  if (!hasLocale(lang)) notFound()
  const dict = await getDictionary(lang)

  return (
    <html lang={lang === 'es' ? 'es-419' : 'en'}>
      <body className="flex flex-col min-h-screen bg-background text-foreground">
        <Header lang={lang} dict={dict} />
        <main className="flex-1 w-full max-w-7xl mx-auto px-4 py-8">
          {children}
        </main>
        <Footer lang={lang} dict={dict} />
      </body>
    </html>
  )
}
```

Key layout requirements:
- `<body>` uses `flex flex-col min-h-screen` so footer sticks to bottom even on short pages
- `<main>` uses `flex-1` to take remaining space, `max-w-7xl mx-auto px-4 py-8` for consistent content width
- Header and Footer receive `lang` and `dict` props from the already-loaded dictionary
- The existing `generateStaticParams` and `generateMetadata` functions remain unchanged
