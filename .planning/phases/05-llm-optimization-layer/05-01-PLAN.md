  <action>
    read_first: src/lib/content/schemas.ts, src/lib/breadcrumbs.ts, src/app/[lang]/page.tsx (existing WebSite JSON-LD pattern)

    1. Install schema-dts: `npm install schema-dts` (per D-05, type-safe JSON-LD generation)

    2. Create `src/lib/jsonld.ts` with these factory functions using schema-dts types for type safety. Import `WithContext` and schema types from 'schema-dts'. Use `WithContext<T>` return types for each builder.

    SITE_URL constant: 'https://www.superfaninfo.com'

    **buildPlaceJsonLd(city: City, lang: Locale): WithContext<Place>**
    - @type: 'Place'
    - name: city.name[lang]
    - description: city.content.overview[lang] (first 300 chars)
    - geo: { @type: 'GeoCoordinates', latitude: city.coordinates.lat, longitude: city.coordinates.lng }
    - url: `${SITE_URL}/${lang}/${lang === 'es' ? 'ciudades' : 'cities'}/${city.slugs[lang]}`
    - containedInPlace: { @type: 'Country', name: country name based on city.country (Mexico/United States/Canada) }

    **buildStadiumJsonLd(stadium: Stadium, cityName: string, lang: Locale): WithContext<StadiumOrArena>**
    - @type: 'StadiumOrArena'
    - name: stadium.name[lang]
    - description: stadium.content.overview[lang] (first 300 chars)
    - maximumAttendeeCapacity: stadium.capacity
    - geo: { @type: 'GeoCoordinates', latitude: stadium.coordinates.lat, longitude: stadium.coordinates.lng }
    - url: `${SITE_URL}/${lang}/${lang === 'es' ? 'estadios' : 'stadiums'}/${stadium.slugs[lang]}`
    - address: { @type: 'PostalAddress', addressLocality: cityName }
    - event: { @type: 'SportsEvent', name: 'FIFA World Cup 2026', startDate: '2026-06-11', endDate: '2026-07-19' }

    **buildSportsTeamJsonLd(team: Team, lang: Locale): WithContext<SportsTeam>**
    - @type: 'SportsTeam'
    - name: team.name[lang]
    - description: team.description[lang]
    - sport: 'Football (Soccer)'
    - url: `${SITE_URL}/${lang}/${lang === 'es' ? 'equipos' : 'teams'}/${team.slugs[lang]}`
    - memberOf: { @type: 'SportsOrganization', name: team.confederation }

    **buildFAQPageJsonLd(faqs: CityFAQ[], lang: Locale): WithContext<FAQPage>**
    - @type: 'FAQPage'
    - mainEntity: array of { @type: 'Question', name: faq.question[lang], acceptedAnswer: { @type: 'Answer', text: faq.answer[lang] } }

    **buildArticleJsonLd(params: { headline: string, description: string, url: string, dateModified: string, lang: Locale }): WithContext<Article>**
    - @type: 'Article'
    - headline: params.headline
    - description: params.description (first 300 chars)
    - url: params.url
    - dateModified: params.dateModified
    - inLanguage: lang === 'es' ? 'es-419' : 'en'
    - publisher: { @type: 'Organization', name: 'SuperFan Mundial 2026', url: SITE_URL }
    - author: { @type: 'Organization', name: 'SuperFan Mundial 2026' }

    **buildOrganizationJsonLd(lang: Locale): WithContext<Organization>**
    - @type: 'Organization'
    - name: 'SuperFan Mundial 2026'
    - url: SITE_URL
    - description: lang-dependent site description
    - sameAs: [] (empty for now, can add social links later)

    **buildItemListJsonLd(items: Array<{ name: string; url: string }>, lang: Locale): WithContext<ItemList>**
    - @type: 'ItemList'
    - itemListElement: items mapped to { @type: 'ListItem', position: index+1, name: item.name, url: item.url }

    **Helper: buildJsonLdScript(data: object): string**
    - Returns JSON.stringify(data) for use with dangerouslySetInnerHTML
    - This is a convenience but optional -- pages can just use JSON.stringify directly

    All factory functions MUST add `'@context': 'https://schema.org'` via WithContext wrapper. Export all functions as named exports.
